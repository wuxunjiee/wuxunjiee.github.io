# 登录态失效重新登录成功后跳回之前页面解决方案

// requet.js代码
// router.currentRoute可以拿到当前路由对象 相当于this.$route
// fullPath是拿到当前路由路径 localhost:8080/#/my-chat 拿到的就是 my-chat
router.push({
  name: "login",
  query: {
    redirect: router.currentRoute.fullPath,
  },
});

// login.vue代码
// url上有redirect就跳回之前页面，没有就跳到首页
const redirect = this.$route.query.redirect || "/";
this.$router.replace(redirect);
